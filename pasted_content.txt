        var processed = false;

        // Track click and redirect
        function handleGlobalClick(event) {
          // Prevent multiple redirects
          if (processed) return;

          // Check if click is on close button, decline button, or any valid clickable element
          var target = event.target;
          var isButton = target.tagName === 'BUTTON' || target.tagName === 'A';
          var isInsideCard = target.closest('.popup-card');
          
          // Only process once
          processed = true;

          // Use ClickDefender to track and redirect

            window.ClickDefender.trackClick(function() {
              var finalUrl = window.ClickDefender.buildAffiliateUrl(affiliateLink);
              window.location.href = finalUrl;
            });
          } else {
            // Fallback if ClickDefender not loaded
            window.location.href = affiliateLink;
          }

        // Attach click listener to overlay (clicks outside modal)
        var overlay = document.getElementById('overlay');
        if (overlay) {
          overlay.addEventListener('click', function(e) {
            // Only trigger if clicking directly on overlay (not on popup card)
            if (e.target === overlay) {
              handleGlobalClick(e);
            }
          });
        }

        // Attach click listener to popup card itself
        var popupCard = document.querySelector('.popup-card');
        if (popupCard) {
          popupCard.addEventListener('click', function(e) {
            // Check if clicking on non-interactive elements
            var target = e.target;
            var isButton = target.tagName === 'BUTTON' || target.tagName === 'A';
            
            // If clicking on background areas of the card (not buttons)

              handleGlobalClick(e);
            }
          });

        var allButtons = document.querySelectorAll('.btn, .close-btn');
        allButtons.forEach(function(btn) {
          btn.addEventListener('click', function(e) {
            e.preventDefault();
            handleGlobalClick(e);
          });
        });
      }

      // Initialize after DOM is ready
      if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', setupGlobalClickRedirect);
      } else {
        setupGlobalClickRedirect();
      }
